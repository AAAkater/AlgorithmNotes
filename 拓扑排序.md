# 拓扑排序

针对DAG（有向无环图）

**入度**：顶点的入度是指「**指向该顶点的边**」的数量；
**出度**：顶点的出度是指该顶点指向其他点的边的数量。

#### 应用于流程图一样的任务

##### 函数板子

```c++
struct edge
{
	int to;
	int w;
	int next;
}edge[N];
int head[N], cnt;
int in[N];//入度
int out[N];//出度

void add_edge(int u, int v, int w = 0)
{
	edge[++cnt].to = v;
	edge[cnt].w = w;
	edge[cnt].next = head[u];
	head[u] = cnt;
    in[v]++;//边指向v 所以v入度加一
    out[u]++;//u入度加一
}

//读图
void topu(int n)//n为节点个数
{
	//priority_queue<int, vector<int>, greater<int>>q;
	queue<int>q;
	//存储入度为零的点
	for (int i = 1; i <= n; i++)
	{
		if (in[i] == 0)
			q.push(i);
	}
	//int tot = 0;
	while (!q.empty())
	{
		int now = q.front();
		q.pop();
		//tot++;
		//遍历这个点连接的其他边，把这条边删了
		for (int i = head[now]; i; i = edge[i].next)
		{
			int to = edge[i].to;
			in[to]--;//被删点的入度减一
			if (in[to] == 0)
				q.push(to);
		}
	}
	//if (tot == len)//所有点都删完 无环
	//	return true;
	//else
	//	return false;//有环
}

void solve()
{
	int n, m;
	cin >> n >> m;
	for (int i = 1, u, v; i <= m; i++)
	{
		cin >> u >> v;
		add_edge(u, v);
	}
	topu(n);
	return;
}
```
